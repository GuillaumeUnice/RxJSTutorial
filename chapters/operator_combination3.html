            let age$ = Rx.Observable.of(27, 25, 29)
            let name$ = Rx.Observable.of('Foo', 'Bar', 'Beer')
            let isDev$ = Rx.Observable.of(true, true, false)
            // Here we compute the created value through a function in last parameter
            Rx.Observable.of(age$, name$, isDev$)
                .zipAll()
                .delay(3000)
